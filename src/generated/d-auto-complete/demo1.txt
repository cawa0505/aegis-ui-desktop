<template>
  <div>
    <d-auto-complete :data-source="dataSource"
                     @search="handleSearch"/>
  </div>
</template>
<script lang="ts">
  import Vue from 'vue';
  import Component from 'vue-class-component';
  import DAutoComplete from '../../index';

  Vue.use(DAutoComplete);
  @Component({
    name: 'Demo1'
  })
  export default class Demo1 extends Vue {
    public dataSource = [];

    public handleSearch(value) {
      this.dataSource = !value ? [] : [
        value,
        value + value,
        value + value + value
      ];
    }
  }
</script>
